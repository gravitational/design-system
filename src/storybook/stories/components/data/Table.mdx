## Usage

Uses [TanStack Table](https://tanstack.com/table) for state and [TanStack Query](https://tanstack.com/query) for data loading.

```tsx
import {
  DataTable,
  TextCell,
  type ColumnDef,
} from '@gravitational/design-system';

interface Row {
  id: number;
  name: string;
}

const columns: ColumnDef<Row, unknown>[] = [
  {
    accessorKey: 'name',
    header: 'Name',
    cell: ({ row }) => <TextCell data={row.original.name} />,
  },
];

function Example() {
  return <DataTable data={data} columns={columns} emptyText="No rows" />;
}
```
