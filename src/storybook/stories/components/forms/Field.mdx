## Usage

```tsx
import { ComposedField, Input } from '@gravitational/design-system';

function Example() {
  return (
    <ComposedField label="Field label" helperText="Helper text">
      <Input placeholder="Input" />
    </ComposedField>
  );
}
```

### Compose mode

For full compositional control over every slot, use `Field.Root` with the slot parts directly.

```tsx
import { Field, Input } from '@gravitational/design-system';

function Example() {
  return (
    <Field.Root required>
      <Field.Label>
        Field label <Field.RequiredIndicator />
      </Field.Label>
      <Input placeholder="Input" />
      <Field.HelperText>Helper text</Field.HelperText>
    </Field.Root>
  );
}
```
