## Usage

```tsx
import { ComposedToggle } from '@gravitational/design-system';
import { useState } from 'react';

function Example() {
  const [on, setOn] = useState(false);
  return (
    <ComposedToggle
      isToggled={on}
      onToggle={() => setOn(v => !v)}
      label="Toggle"
    />
  );
}
```

### Compose mode

For full compositional control over every slot, use `Toggle.Root` with the slot parts directly.

```tsx
import { Toggle } from '@gravitational/design-system';

function Example() {
  return (
    <Toggle.Root>
      <Toggle.HiddenInput />
      <Toggle.Control>
        <Toggle.Thumb />
      </Toggle.Control>
      <Toggle.Label>Toggle</Toggle.Label>
    </Toggle.Root>
  );
}
```
