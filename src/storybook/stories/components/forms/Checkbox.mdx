## Usage

Information about usage here.

```tsx
import { Checkbox, FieldCheckbox } from '@gravitational/design-system';

function Example() {
  return (
    <Checkbox.Root>
      <Checkbox.HiddenInput />
      <Checkbox.Control>
        <Checkbox.Indicator />
      </Checkbox.Control>
      <Checkbox.Label />
    </Checkbox.Root>
  );
}

// FieldCheckbox
function FieldExample() {
  return <FieldCheckbox name="example" label="Example Checkbox" />;
}
```
