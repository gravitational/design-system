## Usage

```tsx
import { ComposedCheckbox } from '@gravitational/design-system';

function Example() {
  return <ComposedCheckbox label="Checkbox" />;
}
```

### Compose mode

For full compositional control over every slot, use `Checkbox.Root` with the slot parts directly.

```tsx
import { Checkbox } from '@gravitational/design-system';

function Example() {
  return (
    <Checkbox.Root>
      <Checkbox.HiddenInput />
      <Checkbox.Control>
        <Checkbox.Indicator />
      </Checkbox.Control>
      <Checkbox.Label>Checkbox</Checkbox.Label>
    </Checkbox.Root>
  );
}
```
