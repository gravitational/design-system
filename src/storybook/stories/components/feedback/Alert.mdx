## Usage

```tsx
import { ComposedAlert } from '@gravitational/design-system';

function Example() {
  return <ComposedAlert kind="danger" title="Something went wrong" />;
}
```

### Compose mode

For full compositional control over every slot, use `Alert.Root` with the slot parts directly.

```tsx
import { Alert, WarningCircleIcon } from '@gravitational/design-system';

function Example() {
  return (
    <Alert.Root kind="warning">
      <Alert.Indicator>
        <WarningCircleIcon />
      </Alert.Indicator>
      <Alert.Content>
        <Alert.Title>Title</Alert.Title>
        <Alert.Description>Description</Alert.Description>
      </Alert.Content>
    </Alert.Root>
  );
}
```
